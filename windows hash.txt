import os
import hashlib

# Path to the Windows folder
windows_folder = r'C:\Windows'

# Initialize an empty list to store file name and hash value pairs
hash_list = []

# Iterate through the Windows folder and find all .exe files
for root, dirs, files in os.walk(windows_folder):
    for file in files:
        if file.endswith('.exe'):
            file_path = os.path.join(root, file)

            # Open the file in binary mode and create an MD5 hash object
            with open(file_path, 'rb') as f:
                hash_object = hashlib.md5()

                # Read the file in chunks and update the hash object
                for chunk in iter(lambda: f.read(4096), b''):
                    hash_object.update(chunk)

                # Get the hexadecimal representation of the hash value
                hash_value = hash_object.hexdigest()

                # Add the file name and hash value to the hash list
                hash_list.append((file_path, hash_value))

# Print the hash list
print(hash_list)
